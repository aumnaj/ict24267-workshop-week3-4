lambda_function.py — ict24267-submit-feedbackPython 3.11
import json
import boto3
import uuid
from datetime import datetime

# สร้าง DynamoDB client
dynamodb = boto3.resource('dynamodb', region_name='ap-southeast-1')
table = dynamodb.Table('ict24267-feedbacks')

def lambda_handler(event, context):
    # Handle CORS preflight (browser ส่ง OPTIONS มาก่อนเสมอ)
    if event.get('requestContext', {}).get('http', {}).get('method') == 'OPTIONS':
        return cors_response(200, {})

    try:
        # แปลง JSON string เป็น dictionary
        if isinstance(event.get('body'), str):
            body = json.loads(event['body'])
        else:
            body = event.get('body', {})

        # ตรวจสอบ field ที่จำเป็น
        required = ['student_name', 'course', 'rating']
        for field in required:
            if not body.get(field):
                return cors_response(400, {'status': 'error', 'message': f'กรุณากรอก {field}'})

        # ตรวจสอบว่า rating อยู่ระหว่าง 1-5
        rating = int(body['rating'])
        if rating < 1 or rating > 5:
            return cors_response(400, {'status': 'error', 'message': 'rating ต้องอยู่ระหว่าง 1-5'})

        # สร้าง feedback_id แบบ unique และบันทึกลง DynamoDB
        feedback_id = f"fb_{uuid.uuid4().hex[:10]}"
        item = {
            'feedback_id':  feedback_id,
            'student_name': body['student_name'],
            'course':       body['course'],
            'rating':       rating,
            'comment':      body.get('comment', ''),
            'timestamp':    datetime.now().isoformat()
        }
        table.put_item(Item=item)

        return cors_response(201, {
            'status':      'success',
            'message':     'บันทึก Feedback แล้ว ขอบคุณ! 🎉',
            'feedback_id': feedback_id
        })

    except Exception as e:
        return cors_response(500, {'status': 'error', 'message': f'เกิดข้อผิดพลาด: {str(e)}'})


def cors_response(status_code, body):
    return {
        'statusCode': status_code,
        'headers': {
            'Content-Type': 'application/json; charset=utf-8',
            'Access-Control-Allow-Origin': '*',
            'Access-Control-Allow-Methods': 'GET,POST,OPTIONS',
            'Access-Control-Allow-Headers': 'Content-Type'
        },
        'body': json.dumps(body, ensure_ascii=False)
    }

lambda_function.py — ict24267-get-feedbacksPython 3.11
import json
import boto3
from boto3.dynamodb.conditions import Attr

dynamodb = boto3.resource('dynamodb', region_name='ap-southeast-1')
table = dynamodb.Table('ict24267-feedbacks')

def lambda_handler(event, context):
    if event.get('requestContext', {}).get('http', {}).get('method') == 'OPTIONS':
        return cors_response(200, {})
    try:
        params = event.get('queryStringParameters') or {}
        course = params.get('course')
        rating = params.get('rating')

        if course or rating:
            filter_expr = None
            if course:
                filter_expr = Attr('course').eq(course)
            if rating:
                rf = Attr('rating').eq(int(rating))
                filter_expr = filter_expr & rf if filter_expr else rf
            response = table.scan(FilterExpression=filter_expr)
        else:
            response = table.scan()

        feedbacks = response.get('Items', [])
        feedbacks.sort(key=lambda x: x.get('timestamp', ''), reverse=True)

        avg_rating = 0
        if feedbacks:
            avg_rating = round(sum(int(f.get('rating', 0)) for f in feedbacks) / len(feedbacks), 2)

        return cors_response(200, {'status': 'success', 'count': len(feedbacks), 'average_rating': avg_rating, 'data': feedbacks})
    except Exception as e:
        return cors_response(500, {'status': 'error', 'message': str(e)})

def cors_response(status_code, body):
    return {'statusCode': status_code, 'headers': {'Content-Type': 'application/json; charset=utf-8', 'Access-Control-Allow-Origin': '*'}, 'body': json.dumps(body, ensure_ascii=False, default=str)}

lambda_function.py — ict24267-daily-summaryPython 3.11
import json
import boto3
from datetime import datetime, date

dynamodb = boto3.resource('dynamodb', region_name='ap-southeast-1')
table = dynamodb.Table('ict24267-feedbacks')

def lambda_handler(event, context):
    print("🔄 เริ่มสรุปสถิติประจำวัน...")
    try:
        response = table.scan()
        feedbacks = response.get('Items', [])
        if not feedbacks:
            print("ℹ️ ยังไม่มี feedback")
            return {'statusCode': 200, 'body': 'ไม่มีข้อมูล feedback'}

        total      = len(feedbacks)
        ratings    = [int(f.get('rating', 0)) for f in feedbacks]
        avg_rating = round(sum(ratings) / total, 2)
        rating_dist = {str(i): ratings.count(i) for i in range(1, 6)}

        course_dist = {}
        for f in feedbacks:
            c = f.get('course', 'ไม่ระบุ')
            course_dist[c] = course_dist.get(c, 0) + 1

        sorted_fb = sorted(feedbacks, key=lambda x: x.get('timestamp', ''), reverse=True)
        recent = [{'student': f.get('student_name'), 'rating': int(f.get('rating', 0)), 'comment': f.get('comment', '')[:50]} for f in sorted_fb[:3]]

        summary = {'report_date': str(date.today()), 'total': total, 'avg_rating': avg_rating, 'rating_dist': rating_dist, 'by_course': course_dist, 'recent_3': recent}

        print(f"📊 {summary['report_date']} | รวม: {total} | เฉลี่ย: {avg_rating}/5")
        print(f"   Rating: {rating_dist} | วิชา: {course_dist}")
        print("✅ สรุปสถิติเสร็จแล้ว")

        return {'statusCode': 200, 'body': json.dumps(summary, ensure_ascii=False, default=str)}
    except Exception as e:
        print(f"❌ Error: {str(e)}")
        raise

lambda_function.py — ict24267-submit-feedback (อัปเดต + SNS)Python 3.11
import json, boto3, uuid
from datetime import datetime

dynamodb = boto3.resource('dynamodb', region_name='ap-southeast-1')
table    = dynamodb.Table('ict24267-feedbacks')
sns      = boto3.client('sns', region_name='ap-southeast-1')

# ⚠️ แทนที่ด้วย SNS Topic ARN จริงของคุณ
SNS_TOPIC_ARN = 'YOUR_SNS_TOPIC_ARN'

def lambda_handler(event, context):
    if event.get('requestContext', {}).get('http', {}).get('method') == 'OPTIONS':
        return cors_response(200, {})
    try:
        body = json.loads(event['body']) if isinstance(event.get('body'), str) else event.get('body', {})
        for field in ['student_name', 'course', 'rating']:
            if not body.get(field):
                return cors_response(400, {'status': 'error', 'message': f'กรุณากรอก {field}'})
        rating = int(body['rating'])
        if not (1 <= rating <= 5):
            return cors_response(400, {'status': 'error', 'message': 'rating ต้องอยู่ระหว่าง 1-5'})

        feedback_id = f"fb_{uuid.uuid4().hex[:10]}"
        item = {'feedback_id': feedback_id, 'student_name': body['student_name'], 'course': body['course'], 'rating': rating, 'comment': body.get('comment', ''), 'timestamp': datetime.now().isoformat()}
        table.put_item(Item=item)

        # ─── ส่ง Email แจ้งเตือนผ่าน SNS ───
        stars = '⭐' * rating
        sns.publish(
            TopicArn=SNS_TOPIC_ARN,
            Subject=f'[ICT24267] Feedback ใหม่จาก {body["student_name"]} {stars}',
            Message=f"""มี Feedback ใหม่!
นักศึกษา: {body['student_name']}
วิชา: {body['course']}
คะแนน: {stars} ({rating}/5)
ความคิดเห็น: {body.get('comment','(ไม่มี)')}
เวลา: {item['timestamp']}
ID: {feedback_id}"""
        )
        return cors_response(201, {'status': 'success', 'message': 'บันทึก Feedback แล้ว ขอบคุณ! 🎉', 'feedback_id': feedback_id})
    except Exception as e:
        return cors_response(500, {'status': 'error', 'message': str(e)})

def cors_response(status_code, body):
    return {'statusCode': status_code, 'headers': {'Content-Type': 'application/json; charset=utf-8', 'Access-Control-Allow-Origin': '*'}, 'body': json.dumps(body, ensure_ascii=False)}
